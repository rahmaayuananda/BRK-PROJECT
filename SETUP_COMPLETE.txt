✅ LOGIN
   When: User berhasil login
   Action: LOGIN
   Target: null
   Example: "User 'John Doe' berhasil login dari IP 192.168.1.1"

✅ CREATE_TOPIC
   When: User membuat topik baru
   Action: CREATE_TOPIC
   Target: {topic_id}
   Example: "User 'John Doe' membuat topik baru dengan judul '...'"

✅ ARCHIVE_TOPIC
   When: User/Admin mengarsipkan atau menutup topik
   Action: ARCHIVE_TOPIC
   Target: {topic_id}
   Example: "User 'John Doe' mengarsipkan topik dengan ID '...'"

✅ MARK_FAQ
   When: Admin menjadikan topik sebagai FAQ
   Action: MARK_FAQ
   Target: {topic_id}
   Example: "Admin 'Admin User' menjadikan topik sebagai FAQ"
